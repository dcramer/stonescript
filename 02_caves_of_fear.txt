var loadouts = import my/loadouts

func equip_melee()
  equipL sword *10
  equipR shield *10

func equip_staff()
  equip staff *5 ice

func equip_ranged()
  equip repeating crossbow 6*

?loc = caves
  ?foe ! boss  // Small enemies
    ?foe.count > 7 & foe.distance > 6 &
    ^item.GetCooldown("blade_of_god") <= 0
      // need some area damage
      equip blade_of_god
      activate R
    :?foe.count > 7 & foe.distance < 10
      // staff stlil sucks
      equip blade_of_god
    :?foe.name = hill | foe.name = snail
      equip_staff()
    :?hp ! maxhp & foe.distance > 12
      // Try to heal while attacking from afar
      loadouts.equip_ranged_healing()
    :?foe.distance > 10
      equip_ranged()
    :
      equip_melee()
  :?foe = boss  // Bolesh, the cunning
    ?foe.distance > 10
      loadouts.equip_dashing()
    :
      equip blade_of_god
      ?item.GetCooldown("blade_of_god") <= 0
        activate R
      // equip_melee()
    ?hp < 10
      activate potion

  loadouts.healwalk()
